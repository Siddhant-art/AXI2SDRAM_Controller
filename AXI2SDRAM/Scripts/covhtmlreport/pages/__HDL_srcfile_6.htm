     2  module async_fifo#(parameter SIZE=32,WIDTH=6,DEPTH=60)(
     3          input rd_clk,
     4          input wr_clk,
     5          input reset_n,
     6          input rd_en,
     7          input wr_en,
     8          input [SIZE-1:0] data_in,
     9          output [SIZE-1:0] data_out,
     10          output fifo_empty,
     11          output fifo_full
     12      );
     13      
     14      reg [SIZE-1:0] data_out_reg;
     15      reg [WIDTH-1:0] rd_ptr;
     16      reg [WIDTH-1:0] wr_ptr;
     17      reg [SIZE-1:0] Data [DEPTH-1:0];
     18      
     19      
     20      reg [WIDTH-1:0] temp_rd_ptr;
     21      reg [WIDTH-1:0] temp_wr_ptr;
     22      reg [WIDTH-1:0] temp_rd_ptr_1;
     23      reg [WIDTH-1:0] temp_rd_ptr_2;
     24      reg [WIDTH-1:0] temp_wr_ptr_1;
     25      reg [WIDTH-1:0] temp_wr_ptr_2;
     26      
     27      //fifo write operation
     28      always@(posedge wr_clk,negedge reset_n)
     29      begin
     30          if(!reset_n) begin
     31              wr_ptr<=0;
     32              Data[wr_ptr+1]<='h00;
     33          end 
     34          else if(wr_en && !fifo_full) begin
     35              Data[wr_ptr]<=data_in;
     36              wr_ptr<=wr_ptr+1;
     37          end
     38      end
     39      
     40      //fifo read operation
     41      always@(posedge rd_clk,negedge reset_n)
     42      begin
     43          if(!reset_n) begin
     44              rd_ptr<=0;
     45              data_out_reg<=0; 
     46          end  
     47          else if(rd_en && !fifo_empty) begin
     48              data_out_reg<=Data[rd_ptr];
     49              rd_ptr<=rd_ptr+1;
     50          end
     51      end
     52      
     53      always@(posedge rd_clk,negedge reset_n)
     54      begin
     55          if(!reset_n) begin
     56               temp_wr_ptr<=0;
     57               temp_wr_ptr_1<=0;
     58               temp_wr_ptr_2<=0;
     59          end  
     60          else begin
     61              temp_wr_ptr<=wr_ptr;
     62              temp_wr_ptr_1<=temp_wr_ptr;
     63              temp_wr_ptr_2<=temp_wr_ptr_1;
     64          end
     65      end
     66      
     67      always@(posedge wr_clk,negedge reset_n)
     68      begin
     69          if(!reset_n) begin
     70               temp_rd_ptr<=0;
     71               temp_rd_ptr_1<=0;
     72               temp_rd_ptr_2<=0;
     73          end  
     74          else begin
     75              temp_rd_ptr<=rd_ptr;
     76              temp_rd_ptr_1<=temp_rd_ptr;
     77              temp_rd_ptr_2<=temp_rd_ptr_1;
     78          end
     79      end
     80      
     81      assign data_out = data_out_reg;
     82      assign fifo_full = (wr_ptr  == temp_rd_ptr_2 + DEPTH);
     83      assign fifo_empty = (temp_wr_ptr_2 == rd_ptr);
     84      
     85  endmodule
