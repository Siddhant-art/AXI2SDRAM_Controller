     139     virtual task drive_axi_write();
     140        
     141            // IDLE State
     142         @(posedge axi_intf.ACLK)
     143  
     144           axi_intf.AWADDR  <= 0;
     145           axi_intf.AWVALID <= 0;
     146           axi_intf.WDATA   <= 0;
     147           axi_intf.WVALID  <= 0;
     148           axi_intf.BREADY  <= 0;
     149  
     150           // Address 
     151         @(posedge axi_intf.ACLK)
     152     
     153           axi_intf.AWADDR  <= axi_pkt.AWADDR;
     154           axi_intf.AWVALID <= axi_pkt.AWVALID;
     155  
     156           // Data
     157         @(posedge axi_intf.ACLK)
     158  
     159           axi_intf.WDATA  <= axi_pkt.WDATA;
     160           axi_intf.WVALID <= axi_pkt.WVALID;
     161           axi_intf.BREADY <= axi_pkt.WVALID;
     162  
     163         @(posedge axi_intf.ACLK)
     164  
     165           axi_intf.AWADDR  <= 0;
     166           axi_intf.AWVALID <= 0;
     167           axi_intf.WDATA   <= 0;
     168           axi_intf.WVALID  <= 0;
     169           axi_intf.BREADY  <= 0;
     170  
     171     endtask
     172  
     173  
     174     virtual task drive_axi_read();
     175      
     176         @(posedge axi_intf.ACLK)
     177     
     178          axi_intf.ARADDR  <= 0;
     179          axi_intf.ARVALID <= 1;
     180          axi_intf.RDATA   <= 0;
     181          axi_intf.RVALID  <= 0;
     182  
     183         @(posedge axi_intf.ACLK)
     184  
     185          axi_intf.ARADDR  <= axi_pkt.ARADDR;
     186          axi_intf.ARVALID <= axi_pkt.ARVALID;
     187  
     188         @(posedge axi_intf.ACLK)
     189  
     190          axi_intf.RDATA   <= axi_pkt.RDATA;
     191          axi_intf.RVALID  <= axi_pkt.RVALID;
     192          axi_intf.RRESP   <= axi_pkt.RRESP;
     193  
     194         @(posedge axi_intf.ACLK)
     195   
     196          axi_intf.ARADDR  <= 0;
     197          axi_intf.ARVALID <= 0;
     198          axi_intf.RDATA   <= 0;
     199          axi_intf.RVALID  <= 0;
     200  
     201     endtask
